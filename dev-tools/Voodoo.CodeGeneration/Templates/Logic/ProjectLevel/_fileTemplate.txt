using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using Voodoo.CodeGeneration.Models;

namespace Voodoo.CodeGeneration.Templates.Logic
{
    public partial class [FILE]Template
    {
        public [FILE]File File { get; set; }
    }

    public class [FILE]File : CodeFile
    {
        public override string GetFileContents()
        {
            return Template.TransformText();
        }

        public override string GetFolder()
        {
            return @"Operations\" + Type.Name;
        }

        public [FILE]Template Template { get; set; }

        public [FILE]File(ProjectFacade project, TypeFacade type)
            : base(project, type)
        {
            this.Template = new [FILE]Template() { File = this };
            this.Name = string.Format("{0}[FILE]", this.Name);
        }

    }
}
